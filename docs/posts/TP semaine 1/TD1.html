library(geosphere)
points <- strsplit(data_ex$geo_point, ",")

min_distance <- 100000000000
site1 <- 0
site2 <- 0


for (i in 1:(length(data_ex$geo_point) - 1)) {
  for (j in (i+1):length(data_ex$geo_point)) {
    distance  <- distHaversine(as.numeric(points[[i]]), as.numeric(points[[j]]))
